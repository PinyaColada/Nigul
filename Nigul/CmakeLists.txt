# Set the minimum required version of CMake
cmake_minimum_required(VERSION 3.12)

# Set the project name
project(Nigul)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 17)

# Set the source directories
set(SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/source)
set(INCLUDE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/dependencies/include)

# Add the source files from both source and include directories
file(GLOB_RECURSE SOURCE_FILES
    ${SOURCE_DIR}/*.cpp
    ${INCLUDE_DIR}/*.cpp
)

# Create the executable
add_executable(Nigul ${SOURCE_FILES})

# Set the include directories for the project
target_include_directories(Nigul PRIVATE ${INCLUDE_DIR})

# Link the glfw3 library to the executable
target_link_libraries(Nigul PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/dependencies/lib/glfw3.lib opengl32.lib)
